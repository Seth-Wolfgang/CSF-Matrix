321  friend std::ostream& operator<<(std::ostream& os, CSF::SparseMatrix<T, indexT, compressionLevel, columnMajor>& mat) {

325  // return os;
326  // }
327  // #endif
328 
329  // create a matrix to store the full matrix representation of the CSF matrix
330  T** matrix = new T * [mat.rows()];
331  for (size_t i = 0; i < mat.rows(); i++) {
332  matrix[i] = new T[mat.cols()];
333  memset(matrix[i], 0, sizeof(matrix[i]) * mat.cols());
334  }
335 
336  // Build the full matrix representation of the the CSF matrix
337  for (size_t i = 0; i < mat.cols(); i++) {
338  for (typename CSF::SparseMatrix<T, indexT, compressionLevel, columnMajor>::InnerIterator it(mat, i); it; ++it) {
339  // std::cout << "it.row(): " << it.row() << " col: " << it.col() << " value: " << it.value() << std::endl;
340  matrix[it.row()][it.col()] = it.value();
341  }
342  }
343 
344 
345  // std::cout << "rows: " << mat.rows() << std::endl;
346  // std::cout << "cols: " << mat.cols() << std::endl;
347 
348  // store all of matrix into the output stream
349  for (size_t i = 0; i < mat.rows(); i++) {
350  for (size_t j = 0; j < mat.cols(); j++) {
351  os << matrix[i][j] << " ";
352  }
353  os << std::endl;
354  }
