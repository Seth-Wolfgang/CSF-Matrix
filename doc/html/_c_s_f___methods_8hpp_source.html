35  catch (std::bad_alloc& e) {

41  #ifdef CSF_PARALLEL
42  #pragma omp parallel for
43  #endif
44  for (size_t i = 0; i < outerDim; i++) {
45  // construct the dictionary
46 
47  // create a std::map that holds value as the key and a vector of indices as the value
48  std::map<T2, std::vector<indexT2>> dict;
49 
50  // check if the column is empty
51  if (outerPtr[i] == outerPtr[i + 1]) {
52

55  endPointers[i] = nullptr;
56 
57  continue;

59 
60  for (indexT2 j = outerPtr[i]; j < outerPtr[i + 1]; j++) {
61

63  if (dict.find(vals[j]) != dict.end()) {
64 
65  // add the index to the vector
66  if constexpr (compressionLevel == 3) {
67 
68  // positive delta encode
69  dict[vals[j]].push_back(innerIndices[j] - dict[vals[j]][1]);
70 
71  // update the last index
72  dict[vals[j]][1] = innerIndices[j];
73 
74  // update the max delta
75  if (dict[vals[j]][dict[vals[j]].size() - 1] > dict[vals[j]][0]) {
76

104  outerByteSize += sizeof(T) + 1 + (pair.second[0] * (pair.second.size() - 2)) + pair.second[0];

134  for (auto& pair : dict) {
135

137  *(T*)helpPtr = (T)pair.first;
138  helpPtr = (T*)helpPtr + 1;
139

143  helpPtr = (uint8_t*)helpPtr + 1;

147  helpPtr = (uint8_t*)helpPtr + 1;

152  if constexpr (compressionLevel == 3) {

155  }

159  case 1:

161  helpPtr = (uint8_t*)helpPtr + 1;
162  break;
163  case 2:
164  *(uint16_t*)helpPtr = (uint16_t)pair.second[k];
165  helpPtr = (uint16_t*)helpPtr + 1;
166  break;
167  case 4:
168  *(uint32_t*)helpPtr = (uint32_t)pair.second[k];
169  helpPtr = (uint32_t*)helpPtr + 1;

171  case 8:
172  *(uint64_t*)helpPtr = (uint64_t)pair.second[k];
173  helpPtr = (uint64_t*)helpPtr + 1;
174  break;
175  }
176  }
177  else {
178  *(indexT*)helpPtr = (indexT)pair.second[k];
179  helpPtr = (indexT*)helpPtr + 1;
180  }
181  }
182 
183  // write the delimiter
184  if constexpr (compressionLevel == 3) {
185  // write a delimiter of the correct width
186  switch (pair.second[0]) {
187  case 1:
188  *(uint8_t*)helpPtr = (uint8_t)DELIM;
189  helpPtr = (uint8_t*)helpPtr + 1;
190  break;
191  case 2:
192  *(uint16_t*)helpPtr = (uint16_t)DELIM;
193  helpPtr = (uint16_t*)helpPtr + 1;
194  break;
195  case 4:

197  helpPtr = (uint32_t*)helpPtr + 1;

201  helpPtr = (uint64_t*)helpPtr + 1;

204  }

207  helpPtr = (indexT*)helpPtr + 1;
208  }
209 
210  // set the end ptr
211  endPointers[i] = helpPtr;
212 
213  } // end dict loop
214 
215  } // end outer loop
216

218 
219  // add up the size of each col and add it to compSize

222  //compSize += (uint8_t *)endPointers[i] - (uint8_t *)data[i];

249  template <typename T, typename indexT, uint8_t compressionLevel, bool columnMajor>
250  uint32_t SparseMatrix<T, indexT, compressionLevel, columnMajor>::encodeVal() {
251  uint8_t byte0 = sizeof(T);
252  uint8_t byte1 = std::is_floating_point<T>::value ? 1 : 0;
253  uint8_t byte2 = std::is_signed_v<T> ? 1 : 0;
254  uint8_t byte3 = columnMajor ? 1 : 0;
255 
256  return (byte3 << 24) | (byte2 << 16) | (byte1 << 8) | byte0;
257  }

260  void SparseMatrix<T, indexT, compressionLevel, columnMajor>::checkVal() {
261  uint8_t byte0 = val_t & 0xFF;
262  uint8_t byte1 = (val_t >> 8) & 0xFF;
263  uint8_t byte2 = (val_t >> 16) & 0xFF;
264  uint8_t byte3 = (val_t >> 24) & 0xFF;
265 
266 
267  assert(byte0 == sizeof(T) && "Value type size does not match");
268 
269  assert(byte1 == std::is_floating_point_v<T> && "Value type is not floating point");
270 
271  assert(byte2 == std::is_signed_v<T> && "Value type is not signed");
272 
273  assert(byte3 == columnMajor && "Major direction does not match");

688  template <typename T, typename indexT, uint8_t compressionLevel, bool columnMajor>
689  CSF::SparseMatrix<T, indexT, compressionLevel, columnMajor> SparseMatrix<T, indexT, compressionLevel, columnMajor>::transpose() {
690  // create an ordered map of unordered maps to store the values
691  std::map<indexT, std::unordered_map<T, std::vector<indexT>>> transposedMap;
692 
693  // iterate over the matrix
694 
695  for (uint32_t i = 0; i < outerDim; ++i) {
696  for (typename SparseMatrix<T, indexT, compressionLevel>::InnerIterator it(*this, i); it; ++it) {
697  // add the value to the map
698  if constexpr (columnMajor) {
699  transposedMap[it.row()][it.value()].push_back(it.col());
700  }
701  else {
702  transposedMap[it.col()][it.value()].push_back(it.row());
703  }
704  }
705  }
706 
707  // if compression level is 3 positive delta encode the vectors of indices
708  if (compressionLevel == 3) {
709  for (auto& row : transposedMap) {
710  for (auto& col : row.second) {
711 
712  // find the max value in the vector
713  size_t max = col.second[0];
714 
715  // delta encode the vector
716  for (uint32_t i = col.second.size() - 1; i > 0; --i) {
717  col.second[i] -= col.second[i - 1];
718  if ((size_t)col.second[i] > max)
719  max = col.second[i];
720  }
721 
722  max = byteWidth(max);
723  // append max to the vector
724  col.second.push_back(max);
725  }
726  }
727  }
728 
729  // create a new matrix passing in transposedMap
730  CSF::SparseMatrix<T, indexT, compressionLevel, columnMajor> transposedMatrix(transposedMap, numCols, numRows);
731 
732  return transposedMatrix;
733  }
